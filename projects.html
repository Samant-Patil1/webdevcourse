CartContext.jsxReact
import { createContext, useContext, useReducer } from 'react';

const CartContext = createContext();

const cartReducer = (state, action) => {
  switch (action.type) {
    case 'ADD_ITEM': {
      const existing = state.items.find(
        item => item.id === action.payload.id && 
                item.variant === action.payload.variant
      );
      if (existing) {
        return {
          ...state,
          items: state.items.map(item =>
            item.id === action.payload.id
              ? { ...item, quantity: item.quantity + 1 }
              : item
          )
        };
      }
      return { ...state, items: [...state.items, action.payload] };
    }
    case 'REMOVE_ITEM':
      return {
        ...state,
        items: state.items.filter(item => item.cartId !== action.payload)
      };
    default:
      return state;
  }
};

export const CartProvider = ({ children }) => {
  const [state, dispatch] = useReducer(cartReducer, { items: [] });
  
  const addToCart = (product, variant, quantity) => {
    dispatch({
      type: 'ADD_ITEM',
      payload: { cartId: `${product.id}-${variant}`, ...product, variant, quantity }
    });
  };
  
  const cartTotal = state.items.reduce(
    (sum, item) => sum + item.price * item.quantity, 0
  );
  
  return (
    <CartContext.Provider value={{ items: state.items, addToCart, cartTotal }}>
      {children}
    </CartContext.Provider>
  );
};

export const useCart = () => useContext(CartContext);

useInfiniteScroll.jsReact
import { useState, useEffect, useCallback } from 'react';

const useInfiniteScroll = (fetchData) => {
  const [data, setData] = useState([]);
  const [page, setPage] = useState(1);
  const [loading, setLoading] = useState(false);
  const [hasMore, setHasMore] = useState(true);

  const loadMore = useCallback(async () => {
    if (loading || !hasMore) return;
    setLoading(true);
    try {
      const result = await fetchData(page);
      if (result.length === 0) setHasMore(false);
      else {
        setData(prev => [...prev, ...result]);
        setPage(prev => prev + 1);
      }
    } finally {
      setLoading(false);
    }
  }, [page, loading, hasMore, fetchData]);

  useEffect(() => {
    const observer = new IntersectionObserver(
      (entries) => {
        if (entries[0].isIntersecting && hasMore && !loading) {
          loadMore();
        }
      },
      { threshold: 0.1 }
    );

    const sentinel = document.getElementById('scroll-sentinel');
    if (sentinel) observer.observe(sentinel);

    return () => observer.disconnect();
  }, [loadMore, hasMore, loading]);

  return { data, loading, hasMore };
};

export default useInfiniteScroll;

weatherApi.jsJS
const API_KEY = 'your_api_key';
const BASE_URL = 'https://api.openweathermap.org/data/2.5';

class WeatherAPI {
  async getWeatherByCoords(lat, lon) {
    const response = await fetch(
      `${BASE_URL}/weather?lat=${lat}&lon=${lon}&appid=${API_KEY}&units=metric`
    );
    if (!response.ok) throw new Error('Failed to fetch');
    return this.formatWeather(await response.json());
  }

  async getWeatherByCity(city) {
    const response = await fetch(
      `${BASE_URL}/weather?q=${city}&appid=${API_KEY}&units=metric`
    );
    if (!response.ok) throw new Error('City not found');
    return this.formatWeather(await response.json());
  }

  async getForecast(lat, lon) {
    const response = await fetch(
      `${BASE_URL}/forecast?lat=${lat}&lon=${lon}&appid=${API_KEY}&units=metric`
    );
    const data = await response.json();
    return this.formatForecast(data.list);
  }

  formatWeather(data) {
    return {
      city: data.name,
      temp: Math.round(data.main.temp),
      humidity: data.main.humidity,
      windSpeed: data.wind.speed,
      description: data.weather[0].description,
      condition: data.weather[0].main,
      icon: data.weather[0].icon
    };
  }
}

export default new WeatherAPI();

useLocalStorage.jsReact
import { useState, useEffect, useCallback } from 'react';

function useLocalStorage(key, initialValue) {
  const [storedValue, setStoredValue] = useState(() => {
    if (typeof window === 'undefined') return initialValue;
    try {
      const item = window.localStorage.getItem(key);
      return item ? JSON.parse(item) : initialValue;
    } catch (error) {
      return initialValue;
    }
  });

  const setValue = useCallback((value) => {
    try {
      const valueToStore = value instanceof Function 
        ? value(storedValue) 
        : value;
      setStoredValue(valueToStore);
      if (typeof window !== 'undefined') {
        window.localStorage.setItem(key, JSON.stringify(valueToStore));
      }
    } catch (error) {
      console.error(error);
    }
  }, [key, storedValue]);

  return [storedValue, setValue];
}

export default useLocalStorage;

const express = require('express');
const http = require('http');
const { Server } = require('socket.io');

const app = express();
const server = http.createServer(app);
const io = new Server(server, {
  cors: { origin: "http://localhost:3000" }
});

const users = new Map();

io.on('connection', (socket) => {
  console.log('User connected:', socket.id);

  // Join room
  socket.on('join', (userId) => {
    users.set(socket.id, { userId, socketId: socket.id });
    socket.broadcast.emit('user-online', userId);
  });

  // Handle message
  socket.on('send-message', (data) => {
    const message = {
      id: Date.now(),
      senderId: data.senderId,
      text: data.text,
      timestamp: new Date()
    };
    
    // Send to recipient
    io.to(data.recipientId).emit('new-message', message);
    
    // Confirm to sender
    socket.emit('message-sent', message);
  });

  // Typing indicator
  socket.on('typing', (data) => {
    socket.to(data.recipientId).emit('user-typing', {
      userId: data.senderId
    });
  });

  // Disconnect
  socket.on('disconnect', () => {
    const user = users.get(socket.id);
    if (user) {
      socket.broadcast.emit('user-offline', user.userId);
      users.delete(socket.id);
    }
  });
});

server.listen(3001, () => {
  console.log('Server running on port 3001');
});

auth.js - JWT MiddlewareNode.js
const jwt = require('jsonwebtoken');
const User = require('../models/User');

const authMiddleware = async (req, res, next) => {
  try {
    const token = req.header('Authorization')?.replace('Bearer ', '');
    
    if (!token) {
      return res.status(401).json({ message: 'No token, access denied' });
    }

    const decoded = jwt.verify(token, process.env.JWT_SECRET);
    const user = await User.findById(decoded.id).select('-password');
    
    if (!user) {
      return res.status(401).json({ message: 'Token is not valid' });
    }

    req.user = user;
    next();
  } catch (error) {
    res.status(401).json({ message: 'Token is not valid' });
  }
};

const adminMiddleware = (req, res, next) => {
  if (req.user && req.user.isAdmin) {
    next();
  } else {
    res.status(403).json({ message: 'Admin access required' });
  }
};

module.exports = { authMiddleware, adminMiddleware };
